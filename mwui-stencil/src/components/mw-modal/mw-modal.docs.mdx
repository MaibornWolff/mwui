```jsx
  <mw-button
      id="open-modal"
      label="Open Modal"
  ></mw-button>
  <mw-modal
    trigger="open-modal" // id of the trigger element that opens the modal
    dismiss-trigger="close-icon" // id of the trigger element that dismisses the modal
    backdrop-dismiss="false"
    size="default"
  >
      <mw-modal-title headline="{headline}" description="{description}">
          <mw-icon icon="close" id="close-icon" size="medium" slot="icon"></mw-icon>
      </mw-modal-title>

      <div>
          all of the modal content
      </div>
      <mw-modal-footer>
          <span slot="end">
            <mw-button label="Action 1" variant="ghost"></mw-button>
            <mw-button label="Action 2"></mw-button>
          </span>
      </mw-modal-footer>
  </mw-modal>
```
